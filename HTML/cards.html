Parallax Tilt Card

Mouse-driven tilt — HTML + CSS + small JS

Tilt Me
<!-- Tilt Card -->
<div class="tilt-card" data-tilt>
  <div class="tilt-inner">Tilt Me</div>
</div>

<!-- CSS -->
.tilt-card { width:260px; height:160px; perspective:900px; }
.tilt-inner { width:100%; height:100%; border-radius:12px; background:linear-gradient(135deg,#667eea,#764ba2); display:flex; align-items:center; justify-content:center; color:#fff; font-weight:700; transition: transform 0.1s ease-out; box-shadow: 0 8px 22px rgba(0,0,0,0.5); }

<!-- JS (small) -->
const el = document.querySelector('[data-tilt]'); // attaches JS to container
el.addEventListener('mousemove', (e) => {
  const rect = el.getBoundingClientRect();
  const x = e.clientX - rect.left - rect.width/2;
  const y = e.clientY - rect.top - rect.height/2;
  const inner = el.querySelector('.tilt-inner');
  inner.style.transform = `rotateY(${x/20}deg) rotateX(${-y/20}deg)`;
});
el.addEventListener('mouseleave', () => inner.style.transform = 'rotateY(0) rotateX(0)'; );
